## Get Transaction

Retrieve detailed information about a specific transaction.

<CodeGroup>
```bash cURL
curl "https://mempool.space/api/v1/tx/15e10745f15593a899cef391191bdd3d7c12412cc4696b7bcb669d0feadc8521"
```

```javascript JavaScript
const txid = '15e10745f15593a899cef391191bdd3d7c12412cc4696b7bcb669d0feadc8521';
const response = await fetch(`https://mempool.space/api/v1/tx/${txid}`);
const transaction = await response.json();
console.log(transaction);
```

```python Python
import requests

txid = '15e10745f15593a899cef391191bdd3d7c12412cc4696b7bcb669d0feadc8521'
response = requests.get(f'https://mempool.space/api/v1/tx/{txid}')
transaction = response.json()
print(transaction)
```
</CodeGroup>

<ParamField path="txId" type="string" required>
  Transaction ID (64 character hex string)
</ParamField>

<ResponseField name="transaction" type="Transaction">
  Transaction object with complete details

    </ResponseField>
    
    <ResponseField name="version" type="number">

    </ResponseField>
    
    <ResponseField name="locktime" type="number">

    </ResponseField>
    
    <ResponseField name="size" type="number">
      Transaction size in bytes
    </ResponseField>
    
    <ResponseField name="weight" type="number">
      Transaction weight (segwit)
    </ResponseField>
    
    <ResponseField name="fee" type="number">
      Transaction fee in satoshis
    </ResponseField>
    
    <ResponseField name="vin" type="Vin[]">
      Array of transaction inputs
      
      <Expandable title="Vin Fields">
        <ResponseField name="txid" type="string">
          Input transaction ID
        </ResponseField>
        
        <ResponseField name="vout" type="number">
          Output index being spent
        </ResponseField>
        
        <ResponseField name="is_coinbase" type="boolean">
          Whether this is a coinbase input
        </ResponseField>
        
        <ResponseField name="scriptsig" type="string">
          Script signature (hex)
        </ResponseField>
        
        <ResponseField name="scriptsig_asm" type="string">
          Script signature (ASM format)
        </ResponseField>
        
        <ResponseField name="sequence" type="number">
          Input sequence number
        </ResponseField>
        
        <ResponseField name="witness" type="string[]">

          Bitcoin address (if applicable)

          Output value in satoshis

    <ResponseField name="status" type="Status">

      <Expandable title="Status Fields">

        <ResponseField name="block_hash" type="string">
          Block hash (if confirmed)
        </ResponseField>
        
        <ResponseField name="block_time" type="number">
          Block timestamp (if confirmed)
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Broadcast Transaction

Broadcast a raw transaction to the network.

<CodeGroup>
```bash cURL
curl -X POST "https://mempool.space/api/v1/tx" \
  -H "Content-Type: text/plain" \
  --data "02000000000101..."
```

```javascript JavaScript
const rawTx = '02000000000101...';
const response = await fetch('https://mempool.space/api/v1/tx/push', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/x-www-form-urlencoded'
  },
  body: `tx=${rawTx}`
});
const txid = await response.text();
console.log(txid);
```

```python Python
import requests

raw_tx = '02000000000101...'
response = requests.post(
    'https://mempool.space/api/v1/tx/push',
    data={'tx': raw_tx}
)
txid = response.text
print(txid)
```
</CodeGroup>

<ParamField body="tx" type="string" required>
  Raw transaction hex string
</ParamField>

<ResponseField name="txid" type="string">
  Transaction ID of the broadcasted transaction
</ResponseField>

<Note>
  Use `POST /api/v1/tx/push` with form data or `POST /api/v1/tx` with raw hex in the body.
</Note>
